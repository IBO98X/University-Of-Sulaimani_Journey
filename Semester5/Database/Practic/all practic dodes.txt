create table employees (
emp_id number(*,0) primary key,
name nvarchar2(50),
age number(2,0),
salary number(5,2),
depart_id number(*,0),
constraint id_fk foreign key(depart_id) references departments(depart_id),
constraint age_check check (age >= 18 and age <= 60)
);
